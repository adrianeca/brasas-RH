function engColor(val){
  var v=String(val||'').toLowerCase().trim()
    .normalize('NFD').replace(/[\u0300-\u036f]/g,''); // remove acentos
  // lookup no ENG_COLOR também normalizado
  var normalized={};
  Object.keys(ENG_COLOR).forEach(function(k){
    var kn=k.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g,'');
    normalized[kn]=ENG_COLOR[k];
  });
  if(normalized[v]) return normalized[v];

  var pos=['excelente','bom','otimo','sempre','sim','positivo','acolhido',
           'abertura','ouvido','respeitoso','leve','fluida','boa','aberta',
           'nada','nunca sairia','por enquanto nada','de nada'];
  var neg=['ruim','pessimo','toxico','nunca','nao','desconsiderada',
           'ignorado','errada','fechada','precisa melhorar',
           'sair','faria sair','falta'];
  var neu=['regular','razoavel','as vezes','depende','talvez','neutro','media'];

  if(pos.some(function(p){ return v.includes(p); })) return 'pos';
  if(neg.some(function(p){ return v.includes(p); })) return 'neg';
  if(neu.some(function(p){ return v.includes(p); })) return 'neu';
  return 'def';
}
function engScore(val){ var k=String(val).toLowerCase().trim(); return ENG_SCORE[k]!==undefined?ENG_SCORE[k]:null; }
function countField(arr,field,multiSelect){
  var counts={};
  arr.forEach(function(r){
    var raw=String(r[field]||'').trim(); if(!raw) return;
    var vals=multiSelect?raw.split(', ').filter(Boolean):[raw];
    vals.forEach(function(v){ counts[v]=(counts[v]||0)+1; });
  });
  return Object.entries(counts).sort(function(a,b){ return b[1]-a[1]; });
}
function avgScoreField(arr,field){
  var vals=arr.map(function(r){ return engScore(r[field]); }).filter(function(v){ return v!==null; });
  return vals.length?vals.reduce(function(s,v){ return s+v; },0)/vals.length:null;
}
function avgScoreFields(arr,fields){
  var scores=fields.map(function(f){ return avgScoreField(arr,f); }).filter(function(v){ return v!==null; });
  return scores.length?scores.reduce(function(s,v){ return s+v; },0)/scores.length:null;
}
function scoreBadge(s,max){
  if(s===null) return '—';
  var label=s/max>=0.75?'Bom':s/max>=0.5?'Regular':'Atenção';
  return s.toFixed(1)+' / '+max+' · '+label;
}
function engBarBlock(arr,field,label,opts){
  opts=opts||{};
  var multiSelect=opts.multiSelect||false,maxWidth=opts.maxWidth||200,colorMap=opts.colorMap||null;
  var entries=countField(arr,field,multiSelect);
  var total=multiSelect?entries.reduce(function(s,e){ return s+e[1]; },0):arr.filter(function(r){ return r[field]&&String(r[field]).trim(); }).length;
  if(!entries.length) return '<div class="eng-question-cell"><div class="eng-q-label">'+esc(label)+'</div><div style="font-size:.78rem;color:var(--gray-400)">Sem respostas</div></div>';
  var maxN=entries[0][1];
  var bars=entries.map(function(e){
    var opt=e[0],n=e[1],pctV=total?Math.round(n/total*100):0,barW=maxN?Math.round(n/maxN*100):0;
    var cls=colorMap?(colorMap[String(opt).toLowerCase().trim()]||'def'):engColor(opt);
    return '<div class="eng-bar-row"><div class="eng-bar-option" style="width:'+maxWidth+'px" title="'+esc(opt)+'">'+esc(opt)+'</div><div class="eng-bar-track"><div class="eng-bar-fill" style="width:'+barW+'%;background:'+(ENG_COLORS[cls]||ENG_COLORS.def)+'"></div></div><div class="eng-bar-n">'+n+'</div><div class="eng-bar-pct">'+pctV+'%</div></div>';
  }).join('');
  return '<div class="eng-question-cell"><div class="eng-q-label">'+esc(label)+'</div>'+bars+'</div>';
}

function doLogin(){

  var pass=document.getElementById('loginPass').value;

    .withFailureHandler(function(e){ btn.disabled=false; btn.textContent='Entrar'; showLoginErr('Erro: '+e.message); })

}

function renderMdOpts(mdId, vals, onchange){
  var optsId=mdId+'OptsInner';
  var inner=document.getElementById(optsId);
  if(!inner) return;
  inner.innerHTML=vals.map(function(u){

function filterMdOpts(mdId){
  var searchEl=document.getElementById(mdId+'Search');
  var div=document.getElementById(mdId+'Opts');
  if(!searchEl||!div) return;
  var q=searchEl.value.toLowerCase().trim();
  var vals=div._vals||[];
  var onchange=div._onchange||'';
  // mantém checked dos que já estavam
  var checked=getMdSel(mdId);
  var filtered=q?vals.filter(function(u){ return u.toLowerCase().includes(q); }):vals;
  renderMdOpts(mdId, filtered, onchange);
  // restaura estado dos checkboxes
  document.querySelectorAll('#'+mdId+'OptsInner input').forEach(function(c){
    if(checked.indexOf(c.value)>=0) c.checked=true;
  });
}
function getMdSel(mdId){
  var vals=[];
  // busca no inner (quando tem busca) ou direto no Opts
  var inner=document.getElementById(mdId+'OptsInner');
  var container=inner||document.getElementById(mdId+'Opts');
  if(!container) return vals;
  container.querySelectorAll('input:checked').forEach(function(c){ if(c.value) vals.push(c.value); });
  return vals;
}

function renderBrindes(){

  var sem=v('f-brinde-sem');
  var selTipo=getMdSel('mdBrindeTipo');
  var selLocal=getMdSel('mdBrindeLocal');

  // expande um registro por (colaborador × ano), para suportar multi-ano
  var gifts=[];
  ALL_BRINDES.forEach(function(e){
    if(!e.ativo) return;
    anosChecked.forEach(function(ano){
      if(!e.giftInfo[ano]) return;
      var fr=isFranquia(e.tipo);

    });
  });

  gifts.sort(function(a,b){
    return (a.ano-b.ano)||String(a.tipo).localeCompare(String(b.tipo))||
           String(a.semestre).localeCompare(String(b.semestre))||
           String(a.unidade).localeCompare(String(b.unidade))||

  });

          '</tr>';

function applyFilters(){
  var fS=v('f-status'),fF=v('f-funcao'),fX=v('f-sexo'),fN=v('f-nome','').trim().toLowerCase();
  var admDe=v('f-adm-de'),admAte=v('f-adm-ate'),desDe=v('f-des-de'),desAte=v('f-des-ate');
  FILTERED=ALL_EMP.filter(function(e){
    if(SELECTED_UNITS.length&&!e.todasUnidades.some(function(u){ return SELECTED_UNITS.includes(u); })) return false;
    if(fS&&e.status!==fS) return false;
    if(fF&&e.funcao!==fF) return false;
    if(fX&&e.sexo!==fX) return false;
    if(fN&&!e.nome.toLowerCase().includes(fN)) return false;
    var admIso=toISO(e.dataAdm);
    if(admDe&&admIso&&admIso<admDe) return false;
    if(admAte&&admIso&&admIso>admAte) return false;
    if(admDe&&!admIso) return false;
    var desISO=toISO(e.dataDeslig);
    if(desDe&&desISO&&desISO<desDe) return false;
    if(desAte&&desISO&&desISO>desAte) return false;
    return true;
  });
  EMP_PAGE=0;
  renderGeral();
}

}

function renderGeral(){
  var ativos=FILTERED.filter(function(e){ return e.ativo; });
  var inativos=FILTERED.filter(function(e){ return !e.ativo; });
  var unitsSet=[...new Set(ativos.map(function(e){ return e.unidade; }).filter(Boolean))];
  set('ins-total',fmt(FILTERED.length)); set('ins-ativos',fmt(ativos.length));
  set('ins-inativos',fmt(inativos.length)); set('ins-unidades',unitsSet.length);
  set('ins-total-d','no filtro atual');
  set('ins-ativos-d',pct(ativos.length,FILTERED.length)+'% do total');
  set('ins-inativos-d',pct(inativos.length,FILTERED.length)+'% do total');
  set('kpi-retencao',(FILTERED.length?pct(ativos.length,FILTERED.length):0)+'%');
  var tempos=ativos.map(function(e){ return e.tenureYears; }).filter(function(t){ return t!==null&&t!==undefined&&!isNaN(t)&&isFinite(t); });
  set('kpi-tempo-medio',tempos.length?avg(tempos).toFixed(1):'—');
  var comTenure=ativos.filter(function(e){ return validNum(e.tenureYears); }).sort(function(a,b){ return b.tenureYears-a.tenureYears; });
  if(comTenure.length){
    var oldest=comTenure[0];
    set('kpi-mais-antigo',oldest.tenureYears+' anos'); set('kpi-mais-antigo-nome',oldest.nome);
    set('kpi-mais-antigo-d',(oldest.funcao||'—')+' · '+(oldest.unidade||'—'));
  } else { set('kpi-mais-antigo','—'); set('kpi-mais-antigo-nome',''); set('kpi-mais-antigo-d',''); }
  var byAdm=[...ativos].filter(function(e){ return e.admYear; }).sort(function(a,b){
    return new Date(b.admYear,(b.admMonth||1)-1,b.admDay||1)-new Date(a.admYear,(a.admMonth||1)-1,a.admDay||1);
  });
  if(byAdm.length){
    var rec=byAdm[0];
    set('kpi-mais-recente',validNum(rec.tenureYears)?rec.tenureYears+' anos':'—');
    set('kpi-mais-recente-nome',rec.nome);
    set('kpi-mais-recente-d',(rec.funcao||'—')+' · '+(rec.unidade||'—'));
  } else { set('kpi-mais-recente','—'); set('kpi-mais-recente-nome',''); set('kpi-mais-recente-d',''); }
  var comPlano=ativos.filter(function(e){ return e.planoSaude&&e.planoSaude!=='-'&&e.planoSaude!==''; }).length;
  set('kpi-plano',fmt(comPlano));
  set('kpi-dep',fmt(ativos.reduce(function(s,e){ return s+(e.numDep||0); },0)));
  renderFuncPyramid(ativos); renderUnitTable(); renderEmpsTable();
}
function renderFuncPyramid(ativos){
  var byFunc=groupBy(ativos,'funcao');
  var all=Object.entries(byFunc).sort(function(a,b){ return b[1].length-a[1].length; });
  var total=ativos.length,maxN=all.length?all[0][1].length:1;
  var shown=all.slice(0,15);
  var rest=all.slice(15);
  if(rest.length>0){
    var outrosEmps=rest.reduce(function(acc,e){ return acc.concat(e[1]); },[]);
    shown.push(['Outros ('+rest.length+' funções)',outrosEmps]);
  }
  set('func-chart-badge',all.length+' funções · '+total+' ativos');
  document.getElementById('funcPyramid').innerHTML=shown.map(function(e,i){
    var fn=e[0],n=e[1].length;
    var pctV=total?(n/total*100).toFixed(1):0;
    var barW=Math.min(100,Math.round(n/maxN*100));
    var color=i<15?PALETTE[i%PALETTE.length]:'#94a3b8';
    return '<div class="func-row"><div class="func-row-label" title="'+esc(fn)+'">'+esc(fn)+'</div><div class="func-bar-wrap"><div class="func-bar-fill" style="width:'+barW+'%;background:'+color+'"></div></div><div class="func-meta"><strong>'+n+'</strong> · '+pctV+'%</div></div>';
  }).join('');
}

function renderEmpsTable(){
  var sc=SORT_EMP;
  var sorted=[...FILTERED].sort(function(a,b){
    var va=a[sc.col],vb=b[sc.col];
    if(sc.col==='dataAdm'||sc.col==='dataDeslig'){ va=toISO(va)||''; vb=toISO(vb)||''; }
    if(va===null||va===undefined) va=''; if(vb===null||vb===undefined) vb='';
    return(typeof va==='string'?va.localeCompare(vb):va-vb)*sc.dir;
  });

function empAdm(e){ if(!e.admYear) return null; return new Date(e.admYear,(e.admMonth||1)-1,e.admDay||1); }
function empDeslig(e){ if(!e.desligYear) return null; return new Date(e.desligYear,(e.desligMonth||1)-1,e.desligDay||1); }
function isActiveAt(e,d){
  var ad=empAdm(e),dd=empDeslig(e);
  if(!ad||ad>d) return false;
  if(dd&&dd<=d) return false;
  return true;
}
function computePeriod(emps,from,to){
  var saldoInicio=emps.filter(function(e){ return isActiveAt(e,from); }).length;
  var saldoFim=emps.filter(function(e){ return isActiveAt(e,to); }).length;
  var contrat=emps.filter(function(e){ return e.admYear&&empAdm(e)>=from&&empAdm(e)<to; }).length;
  var demiss=emps.filter(function(e){ return e.desligYear&&empDeslig(e)>=from&&empDeslig(e)<to; }).length;
  var rot=contrat+demiss,variacao=saldoFim-saldoInicio;
  var media=(saldoInicio+saldoFim)/2||1;
  var taxaVar=saldoInicio?variacao/saldoInicio:0;
  var turnover=rot/media,taxaAdm=contrat/media,taxaDem=demiss/media;
  var deslig6m=emps.filter(function(e){
    if(!e.desligYear) return false;
    var dd=empDeslig(e); if(!dd||dd<from||dd>=to) return false;
    var ad=empAdm(e); if(!ad) return false;
    return (dd-ad)<6*30.44*24*3600*1000;
  }).length;
  return {saldoInicio,contrat,demiss,rot,saldoFim,variacao,taxaVar,turnover,taxaAdm,taxaDem,deslig6m,taxa6m:deslig6m/media};
}

function renderProfNivelTable(profs){
  // expande professores com unidade secundária, excluindo unidades não-operacionais
  var EXCL_PROF_UNITS=['EDITORA','MÉTODOS','EC NEW'];
  var rows=[];
  profs.forEach(function(e){
    if(EXCL_PROF_UNITS.indexOf(e.unidade)<0) rows.push({unidade:e.unidade, nivel:e.nivel});
    if(e.unidadeSec&&EXCL_PROF_UNITS.indexOf(e.unidadeSec)<0) rows.push({unidade:e.unidadeSec, nivel:e.nivel});
  });

  var niveis=NIVEL_ORDER.filter(function(n){
    return rows.some(function(r){ return r.nivel===n; });
  });
  var unidades=[...new Set(rows.map(function(r){ return r.unidade; }))].sort();

  var hStyle='style="background:var(--navy-700);color:#fff"';
  var header='<tr '+hStyle+'><th '+hStyle+'>Unidade</th>'+
    niveis.map(function(n){ return '<th class="center" '+hStyle+'>'+n+'</th>'; }).join('')+
    '<th class="num" '+hStyle+'>Total geral</th></tr>';

  var body=unidades.map(function(u,i){
    var rowBg=i%2===0?'background:var(--gray-50)':'';
    var uRows=rows.filter(function(r){ return r.unidade===u; });
    var total=uRows.length;
    var cells=niveis.map(function(n){
      var c=uRows.filter(function(r){ return r.nivel===n; }).length;
      return '<td class="center">'+(c||'-')+'</td>';
    }).join('');
    return '<tr style="'+rowBg+'"><td><strong>'+esc(u)+'</strong></td>'+cells+'<td class="num"><strong>'+total+'</strong></td></tr>';
  }).join('');

  // linha de totais
  var totRow='<tr class="row-total"><td>Total geral</td>'+
    niveis.map(function(n){
      var c=rows.filter(function(r){ return r.nivel===n; }).length;
      return '<td class="center">'+c+'</td>';
    }).join('')+
    '<td class="num">'+rows.length+'</td></tr>';

  var el=document.getElementById('tblProfNivel');
  if(el) el.innerHTML=header+body+totRow;
}

function renderTurmasChart(){
  var fUn=v('f-prof-unidade');
  var data=fUn?ALL_TURMAS.filter(function(r){ return r.unidade===fUn; }):ALL_TURMAS;

  var counts={},apelidoMap={},unidadeMap={},nivelMap={};
  data.forEach(function(r){
    var key=r.chaveMatricula||r.apelido;
    counts[key]=(counts[key]||0)+1;
    if(!apelidoMap[key]) apelidoMap[key]=r.apelido;
    if(!unidadeMap[key]) unidadeMap[key]=r.unidade||'';
    if(!nivelMap[key]&&r.nivel) nivelMap[key]=r.nivel;
  });

  var allEntries=Object.entries(counts);

  // Média calculada sempre sobre TODOS os professores (ignora filtro de unidade)
  var globalCounts={};
  ALL_TURMAS.forEach(function(r){
    var key=r.chaveMatricula||r.apelido;
    globalCounts[key]=(globalCounts[key]||0)+1;
  });
  var globalVals=Object.values(globalCounts);
  var avg=globalVals.length>0?Math.round((globalVals.reduce(function(s,v){ return s+v; },0)/globalVals.length)*10)/10:0;

  // Top 15 mais turmas (decrescente)
  var top15=allEntries.slice().sort(function(a,b){ return b[1]-a[1]; }).slice(0,15);
  var top15keys=top15.reduce(function(m,e){ m[e[0]]=true; return m; },{});

  // Bottom 15 menos turmas (crescente, sem overlap com top15)
  var bot15=allEntries.slice().sort(function(a,b){ return a[1]-b[1]; })
    .filter(function(e){ return !top15keys[e[0]]; }).slice(0,15);

  var combined=top15.concat(bot15);
  var labels=combined.map(function(e){
    var key=e[0];
    var ap=apelidoMap[key]||key;
    var un=unidadeMap[key]||'';
    var nv=nivelMap[key]||'';
    var line2=un+(un&&nv?' · ':'')+nv;
    return line2?[ap,line2]:ap;
  });
  var n=combined.length;

  // Dois datasets separados para ter legenda correta
  var topVals=combined.map(function(e,i){ return i<top15.length?e[1]:null; });
  var botVals=combined.map(function(e,i){ return i>=top15.length?e[1]:null; });
  var avgVals=labels.map(function(){ return avg; });

  var w=document.getElementById('wrapTurmasComb');
  if(w) w.style.height='440px';
  var ctx=document.getElementById('chartTurmasComb');
  if(ctx){
    if(CHARTS['chartTurmasComb']){ CHARTS['chartTurmasComb'].destroy(); delete CHARTS['chartTurmasComb']; }
    CHARTS['chartTurmasComb']=new Chart(ctx,{
      data:{
        labels:labels,
        datasets:[
          {type:'bar',label:'Mais turmas',data:topVals,backgroundColor:'#2a4d76',borderRadius:4,barThickness:16,order:2},
          {type:'bar',label:'Menos turmas',data:botVals,backgroundColor:'#c9a227',borderRadius:4,barThickness:16,order:2},
          {type:'line',label:'Média BRASAS ('+avg+')',data:avgVals,borderColor:'#e05c2a',borderWidth:2,
           borderDash:[6,4],pointRadius:0,fill:false,order:1,tension:0}
        ]
      },
      options:{
        responsive:true,maintainAspectRatio:false,
        plugins:{
          legend:{display:true,position:'top',labels:{font:{family:'DM Sans',size:11},boxWidth:12,padding:12}},
          tooltip:{callbacks:{label:function(c){ return c.raw!=null?' '+c.dataset.label+': '+fmt(c.raw):''; }}}
        },
        scales:{
          x:{grid:{color:'rgba(0,0,0,.04)'},ticks:{font:{family:'DM Sans',size:10},maxRotation:90,minRotation:45,autoSkip:false}},
          y:{grid:{color:'rgba(0,0,0,.04)'},ticks:{font:{family:'DM Sans',size:11}},beginAtZero:true}
        }
      }
    });
  }
}

function initHistoricoFilters(){
  var EXCL=['EDITORA','MÉTODOS','EC NEW'];
  // Unidades vêm das colunas de acesso dos professores ativos em ALL_EMP
  var profs=ALL_EMP.filter(function(e){
    return e.ativo && e.funcao && e.funcao.toUpperCase().includes('PROFESSOR');
  });
  var units=[...new Set(
    profs.flatMap(function(e){ return [e.unidadeAcesso,e.unidadeSecAcesso]; }).filter(Boolean)
  )].sort();
  var sel=document.getElementById('hist-unidade');
  if(sel){
    while(sel.options.length>1) sel.remove(1);
    units.filter(function(u){ return EXCL.indexOf(u)<0; })
      .forEach(function(u){ sel.appendChild(new Option(u,u)); });
  }
}

function mesLabel_(mesStr){
  // 'MM/YYYY' → 'Mmm/YY'
  var p=mesStr.split('/');
  return MONTHS_PT[parseInt(p[0],10)-1]+'/'+p[1].slice(2);
}

function onHistUnidadeChange(){
  var fUn=v('hist-unidade');
  var sel=document.getElementById('hist-professor');
  if(!sel) return;
  while(sel.options.length>1) sel.remove(1);

  // Fonte de verdade: professores ativos de ALL_EMP filtrados pelas colunas de acesso
  var profs=ALL_EMP.filter(function(e){
    return e.ativo && e.funcao && e.funcao.toUpperCase().includes('PROFESSOR');
  });
  if(fUn){
    profs=profs.filter(function(e){
      return e.unidadeAcesso===fUn || e.unidadeSecAcesso===fUn;
    });
  }

  // Índice apelido → chaveMatricula construído a partir de todos os meses do histórico
  var apelidoToChave={};
  ALL_HISTORICO.forEach(function(m){
    m.rows.forEach(function(r){
      if(!apelidoToChave[r.apelido]) apelidoToChave[r.apelido]=r.chaveMatricula;
    });
  });

  var seen={};
  profs.slice().sort(function(a,b){ return a.apelido.localeCompare(b.apelido); })
    .forEach(function(e){
      var ap=e.apelido; if(!ap||seen[ap]) return;
      seen[ap]=true;
      // valor = chaveMatricula para match no gráfico; fallback = apelido
      sel.appendChild(new Option(ap, apelidoToChave[ap]||ap));
    });

  renderHistorico();
}

    hKpi('Turmas (mês atual)',curT,dl(diffT)+' vs. '+firstMes,diffT>=0)+
    hKpi('Alunos (mês atual)',curA,dl(diffA)+' vs. '+firstMes,diffA>=0)+
    hKpi('vs. Média BRASAS',dl(vsAvg)+' turmas','média atual: '+avgCurT,vsAvg>=0);

  // Gráfico
  var ctx=document.getElementById('chartHistorico');
  if(!ctx) return;
  if(CHARTS['chartHistorico']){ CHARTS['chartHistorico'].destroy(); delete CHARTS['chartHistorico']; }

        {type:'line',label:'Média BRASAS turmas',data:avgTurmas,borderColor:'#89afd4',borderWidth:1.5,borderDash:[5,4],pointRadius:2,fill:false,tension:.3,yAxisID:'yT',order:2},
        {type:'line',label:'Alunos do professor',data:profAlunos,borderColor:'#0d9488',backgroundColor:'rgba(13,148,136,.08)',borderWidth:2.5,pointRadius:4,fill:true,tension:.3,yAxisID:'yA',order:1},

  });

// CHART.JS HELPER
function mkChart(canvasId,type,labels,data,label,opts){
  opts=opts||{};
  var ctx=document.getElementById(canvasId); if(!ctx) return;
  if(CHARTS[canvasId]){ CHARTS[canvasId].destroy(); delete CHARTS[canvasId]; }
  var colors=opts.colors||PALETTE;
  var bgColors=type==='doughnut'
    ?labels.map(function(_,i){ return colors[i%colors.length]; })
    :(colors.length===1?labels.map(function(){ return colors[0]; }):labels.map(function(_,i){ return colors[i%colors.length]; }));
  CHARTS[canvasId]=new Chart(ctx,{
    type,
    data:{labels,datasets:[{label,data,backgroundColor:bgColors,borderColor:type==='doughnut'?'#fff':bgColors,borderWidth:type==='doughnut'?2:0,borderRadius:type==='bar'?4:0,barThickness:opts.barThickness}]},
    options:{

function loadHorasData(){

function initHorasFilters(){
  var anos=[...new Set(ALL_HORAS.map(function(r){ return r.ano; }).filter(Boolean))].sort(function(a,b){ return Number(b)-Number(a); });
  var meses=[...new Set(ALL_HORAS.map(function(r){ return r.mes; }).filter(Boolean))].sort();
  var units=[...new Set(ALL_HORAS.map(function(r){ return r.unidade; }).filter(Boolean))].sort();
  var niveis=[...new Set(ALL_HORAS.map(function(r){ return r.nivel; }).filter(Boolean))].sort();

  ['h-ano','f2-ano'].forEach(function(id){
    var el=document.getElementById(id);
    if(el) anos.forEach(function(a){ el.appendChild(new Option(a,a)); });
  });
  ['h-mes','f2-mes'].forEach(function(id){
    var el=document.getElementById(id);
    if(el) meses.forEach(function(m){ el.appendChild(new Option(m,m)); });
  });
  ['h-unidade','f2-unidade'].forEach(function(id){
    var el=document.getElementById(id);
    if(el) units.forEach(function(u){ el.appendChild(new Option(u,u)); });
  });
  var hN=document.getElementById('h-nivel');
  if(hN) niveis.forEach(function(n){ hN.appendChild(new Option(n,n)); });

  // seleciona o ano mais recente disponível
  var anoAtual=String(new Date().getFullYear());
  ['h-ano','f2-ano'].forEach(function(id){
    var el=document.getElementById(id);
    if(!el||!anos.length) return;
    el.value=anos.indexOf(anoAtual)>=0?anoAtual:anos[0];
  });
}

function filteredHoras(){
  var ano=v('h-ano'),mes=v('h-mes'),uni=v('h-unidade'),niv=v('h-nivel');
  var nome=v('h-nome','').trim().toLowerCase();
  return ALL_HORAS.filter(function(r){
    if(ano&&r.ano!==ano) return false;
    if(mes&&r.mes!==mes) return false;
    if(uni&&r.unidade!==uni) return false;
    if(niv&&r.nivel!==niv) return false;
    if(nome&&!r.nome.toLowerCase().includes(nome)&&!r.apelido.toLowerCase().includes(nome)) return false;
    return true;
  });
}

function filteredFaltas(){
  var ano=v('f2-ano'),mes=v('f2-mes'),uni=v('f2-unidade');
  return ALL_HORAS.filter(function(r){
    if(ano&&r.ano!==ano) return false;
    if(mes&&r.mes!==mes) return false;
    if(uni&&r.unidade!==uni) return false;
    return true;
  });
}

    profMap[k].fDesc+=r.fDesc;
    profMap[k].fAbon+=r.fAbon;
    profMap[k].custoFDesc+=r.custoFDesc;
    profMap[k].custoFAbon+=r.custoFAbon;
  });
  var profs=Object.values(profMap);

  var rankDesc=[...profs].filter(function(p){ return p.fDesc>0; }).sort(function(a,b){ return b.fDesc-a.fDesc; }).slice(0,20);

var ALL_COPA=[];
var COPA_PAGE=0;
var COPA_PAGE_SIZE=50;

function loadCopaData(){
  google.script.run
    .withSuccessHandler(function(r){
      hideTabLoading('copa');
      if(!r.ok){ console.warn('Copa:',r.error); return; }
      ALL_COPA=r.rows||[];
      initCopaFilters();
      renderCopa();
    })
    .withFailureHandler(function(e){ console.warn('Copa falhou:',e.message); })
    .getCopaData(TOKEN);
}

function initCopaFilters(){
  var units=[...new Set(ALL_COPA.map(function(r){ return r.unidade; }).filter(Boolean))].sort();
  var meses=[...new Set(ALL_COPA.map(function(r){ return r.mesAno; }).filter(Boolean))].sort(function(a,b){
    var pa=a.split('/'),pb=b.split('/');
    return (Number(pb[1])-Number(pa[1]))||( Number(pb[0])-Number(pa[0]));
  });
  var cpU=document.getElementById('cp-unidade');
  if(cpU) units.forEach(function(u){ cpU.appendChild(new Option(u,u)); });
  var cpM=document.getElementById('cp-mes');
  if(cpM) meses.forEach(function(m){ cpM.appendChild(new Option(m,m)); });
}

function filteredCopa(){
  var uni=v('cp-unidade'),mes=v('cp-mes'),nome=v('cp-nome','').trim().toLowerCase();
  return ALL_COPA.filter(function(r){
    if(uni&&r.unidade!==uni) return false;
    if(mes&&r.mesAno!==mes) return false;
    if(nome&&!r.beneficiario.toLowerCase().includes(nome)) return false;
    return true;
  });
}

function renderCopa(){
  var data=filteredCopa();
  var totalValor=data.reduce(function(s,r){ return s+r.valor; },0);
  var totalGrat=data.reduce(function(s,r){ return s+r.gratuidades; },0);

  document.getElementById('copa-kpis').innerHTML=
    '<div class="kpi-card highlight"><div class="kpi-label">Total a Cobrar</div><div class="kpi-value">'+fmtBRL(totalValor)+'</div><div class="kpi-sub">no filtro</div></div>'+
    '<div class="kpi-card"><div class="kpi-label">Total Gratuidades</div><div class="kpi-value">'+fmtBRL(totalGrat)+'</div><div class="kpi-sub">coparticipações gratuitas</div></div>'+
    '<div class="kpi-card"><div class="kpi-label">Beneficiários</div><div class="kpi-value">'+new Set(data.map(function(r){ return r.beneficiario; })).size+'</div><div class="kpi-sub">distintos</div></div>'+
    '<div class="kpi-card"><div class="kpi-label">Registros</div><div class="kpi-value">'+data.length+'</div><div class="kpi-sub">linhas no filtro</div></div>';

  set('copa-badge', data.length+' registros');

  // agrupa por unidade > mesAno > matricula para montar tabela hierárquica
  var html='';
  var byUnidade=groupBy(data,'unidade');
  Object.keys(byUnidade).sort().forEach(function(uni){
    var rowsUni=byUnidade[uni];
    var byMes=groupBy(rowsUni,'mesAno');
    Object.keys(byMes).sort().forEach(function(mes){
      var rowsMes=byMes[mes];
      var byMat=groupBy(rowsMes,'matricula');
      var firstMes=true;
      Object.keys(byMat).sort().forEach(function(mat){
        var rowsMat=byMat[mat];
        var totalMat=rowsMat.reduce(function(s,r){ return s+r.valor; },0);
        var firstMat=true;
        rowsMat.forEach(function(r){
          html+='<tr>'+
            '<td>'+(firstMes?esc(uni):'')+'</td>'+
            '<td>'+(firstMes?esc(mes):'')+'</td>'+
            '<td>'+(firstMat?esc(mat):'')+'</td>'+
            '<td>'+esc(String(r.beneficiario||'').toUpperCase())+'</td>'+
            '<td class="num">'+fmtBRL(r.valor)+'</td>'+
            '</tr>';
          firstMes=false; firstMat=false;
        });
        html+='<tr style="background:var(--gray-50)">'+
          '<td></td><td></td><td></td>'+
          '<td style="font-weight:700">Total</td>'+
          '<td class="num" style="font-weight:700">'+fmtBRL(totalMat)+'</td>'+
          '</tr>';
      });
      var totalMesVal=rowsMes.reduce(function(s,r){ return s+r.valor; },0);
      html+='<tr class="row-total">'+
        '<td></td><td></td>'+
        '<td colspan="2">Total '+esc(mes)+'</td>'+
        '<td class="num">'+fmtBRL(totalMesVal)+'</td>'+
        '</tr>';
    });
  });
 window._copaRows=html;
  COPA_PAGE=0;
  renderCopaPage();

  // tabela gratuidades — apenas quem tem gratuidades > 0
  var gratData=data.filter(function(r){ return r.gratuidades>0; });
  set('grat-badge', gratData.length+' registros');
  document.getElementById('tblGratBody').innerHTML=gratData.length
    ?gratData.map(function(r){
        return '<tr>'+
          '<td>'+esc(r.unidade)+'</td>'+
          '<td>'+esc(r.mesAno)+'</td>'+
          '<td>'+esc(r.matricula)+'</td>'+
          '<td>'+esc(String(r.beneficiario||'').toUpperCase())+'</td>'+
          '<td class="num" style="font-weight:700">'+fmtBRL(r.gratuidades)+'</td>'+
          '</tr>';
      }).join('')
    :'<tr><td colspan="5" class="table-empty">Sem gratuidades no filtro.</td></tr>';
}

function renderCopaPage(){
  var allRows=window._copaRows||'';
  var parser=new DOMParser();
  var doc=parser.parseFromString('<table>'+allRows+'</table>','text/html');
  var trs=[...doc.querySelectorAll('tr')];
  var total=trs.length;
  var pages=Math.max(1,Math.ceil(total/COPA_PAGE_SIZE));
  if(COPA_PAGE>=pages) COPA_PAGE=pages-1;
  var start=COPA_PAGE*COPA_PAGE_SIZE;
  var shown=trs.slice(start,start+COPA_PAGE_SIZE).map(function(tr){ return tr.outerHTML; }).join('');
  document.getElementById('tblCopaBody').innerHTML=shown||'<tr><td colspan="5" class="table-empty">Sem dados.</td></tr>';
  var info=document.getElementById('copa-pag-info');
  if(info) info.textContent='Exibindo '+(start+1)+'–'+Math.min(start+COPA_PAGE_SIZE,total)+' de '+total+' linhas';
  var prev=document.getElementById('copa-pag-prev'),next=document.getElementById('copa-pag-next');
  if(prev) prev.disabled=COPA_PAGE===0;
  if(next) next.disabled=COPA_PAGE>=pages-1;
}
function pagCopa(dir){ COPA_PAGE+=dir; renderCopaPage(); }

// UTILS
function v(id,def){ def=def||''; var el=document.getElementById(id); return el?(el.value||def):def; }
function set(id,val){ var el=document.getElementById(id); if(el) el.textContent=val; }
function fmt(n){ return(n===null||n===undefined)?'—':Number(n).toLocaleString('pt-BR'); }
function pct(a,b){ return b?Math.round(a/b*100):0; }
function avg(arr){ return arr.length?arr.reduce(function(s,n){ return s+n; },0)/arr.length:0; }
function esc(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }
function validNum(x){ return x!==null&&x!==undefined&&!isNaN(x)&&isFinite(x); }
function fmtNow(){ var n=new Date(); return n.toLocaleDateString('pt-BR')+' às '+n.toLocaleTimeString('pt-BR',{hour:'2-digit',minute:'2-digit'}); }
function groupBy(arr,key){ return arr.reduce(function(acc,e){ var k=e[key]||'(não informado)'; if(!acc[k]) acc[k]=[]; acc[k].push(e); return acc; },{}); }
function toISO(str){ if(!str||str==='—') return ''; var m=str.match(/^(\d{2})\/(\d{2})\/(\d{4})$/); return m?m[3]+'-'+m[2]+'-'+m[1]:''; }
function fmtDateBr(val){
  if(!val) return '—';
  var d=new Date(val);
  if(isNaN(d.getTime())) return String(val);
  return String(d.getDate()).padStart(2,'0')+'/'+String(d.getMonth()+1).padStart(2,'0')+'/'+d.getFullYear();
}
